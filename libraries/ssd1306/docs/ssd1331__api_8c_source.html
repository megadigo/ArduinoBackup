46 void ssd1331_drawLine(uint8_t x1, uint8_t y1, uint8_t x2, uint8_t y2, uint16_t color)
47 {
48  ssd1306_intf.start();
49  ssd1306_spiDataMode(0);
50  ssd1306_intf.send(SSD1331_DRAWLINE);
51  ssd1306_intf.send(x1);
52  ssd1306_intf.send(y1);
53  ssd1306_intf.send(x2);
54  ssd1306_intf.send(y2);
55  ssd1306_intf.send( (color & 0x03) << 4 );
56  ssd1306_intf.send( (color & 0x1C) << 2 );
57  ssd1306_intf.send( (color & 0xE0) >> 2 );
58  ssd1306_intf.stop();
59 }

62 {
63  uint16_t count = w * h;
64  ssd1306_lcd.set_block(x, y, w);
